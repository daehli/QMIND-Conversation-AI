```bash title="Docker-compose file" subtitle="Environment variable"
version: '3'
services:
  botpress:
    image: botpress/server
    environment:
      - DATABASE_URL=postgres://postgres:secretpw@postgres:5435/botpress_db
      - EXTERNAL_URL=https://learnbotpress.com

  postgres:
    image: postgres:11.2-alpine
    environment:
      PGPORT: 5435
      POSTGRES_DB: botpress_db
      POSTGRES_PASSWORD: secretpw
      POSTGRES_USER: postgres

  nginx:
    image: jonasal/nginx-certbot:latest
    environment:
      - CERTBOT_EMAIL=YOUR_EMAIL@DOMAIN.COM

```

```bash title="Docker-compose file" subtitle="Ports"
version: '3'
services:
  botpress:
    image: botpress/server
    environment:
      - DATABASE_URL=postgres://postgres:secretpw@postgres:5435/botpress_db
      - EXTERNAL_URL=https://learnbotpress.com

  postgres:
    image: postgres:11.2-alpine
    environment:
      PGPORT: 5435
      POSTGRES_DB: botpress_db
      POSTGRES_PASSWORD: secretpw
      POSTGRES_USER: postgres

  nginx:
    image: jonasal/nginx-certbot:latest
    ports:
      - 80:80
      - 443:443
    environment:
      - CERTBOT_EMAIL=YOUR_EMAIL@DOMAIN.COM

```

```bash title="Docker-compose file" subtitle="Volumes"
version: '3'
services:
  botpress:
    image: botpress/server
    environment:
      - DATABASE_URL=postgres://postgres:secretpw@postgres:5435/botpress_db
      - EXTERNAL_URL=https://learnbotpress.com
    volumes:
      - ./build/botpress/data:/botpress/data


  postgres:
    image: postgres:11.2-alpine
    environment:
      PGPORT: 5435
      POSTGRES_DB: botpress_db
      POSTGRES_PASSWORD: secretpw
      POSTGRES_USER: postgres
    volumes:
      - pgdata:/var/lib/postgresql/data

  nginx:
    image: jonasal/nginx-certbot:latest
    ports:
      - 80:80
      - 443:443
    environment:
      - CERTBOT_EMAIL=YOUR_EMAIL@DOMAINE.COM
    volumes:
      - nginx_secrets:/etc/letsencrypt
      - ./nginx/user_conf.d:/etc/nginx/user_conf.d

```
